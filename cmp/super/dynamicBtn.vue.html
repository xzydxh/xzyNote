    <base-button
      v-for="(item, index) in NormalBtn"
      :key="index"
      :disabled="[, false, true][item.diseabledFlag] && ids.length == 0"
      @click="runBtnEvent(item.script, item.params)"
      >{{ item.permissionName }}
</base-button>

    # 下拉按钮

    • 使用效果：

    • 代码：
    <base-drop-button
      v-for="(item, index) in SpecialBtn"
      :key="index"
      :buttonTitle="item.permissionName"
      :dropItems="item.flowPageList"
      @click.native="entityTypeHandle(item.flowPageList, item)"
      @handleDropItem="handleTypeDropItem(item.flowPageList[index], item)"
 ></base-drop-button>

    //流程新建
    entityTypeHandle(arr, item) {
      if (item.permissionCode == "object_button_import") {      // 导入按钮
        // this.runBtnEvent(item.script, item.params);
        return;
      } else {
        if (arr.length == 0) {
          this.$message.warning("暂无配置流程！");
        }
      }
    },
        
        
      //新建流程类实体&&导入
    handleTypeDropItem(item, row) {
      // 导入
      if (row.permissionCode == "object_button_import") {
        this.runBtnEvent(row.script, row.params);
        this.$store.state.common.processData.nodeId = item.nodeId;
        this.$store.state.common.processData.flowId = item.flowId;
        return;
      } else {
        this.$store.state.common.processData.nodeId = "";
        this.$store.state.common.processData.flowId = "";
      }
        
        // 跳转到对应新建流程页
      this.$router.push({
        path: "/entityobject/operation/add",
        query: {
          entityId: this.entity.id,
          systemIdentification: this.entity.systemIdentification,
          path: this.$route.path,
          url: this.$route.path,
          queryData: JSON.stringify(this.$route.query),
          //流程类实体传参
          pageId: item.id,
          flowId: item.flowId,
          node_id: item.nodeId,
        },
      });
    },

        ref="aos-button"
        v-show="isDelete != 1"
        :ids="ids"

        :editEntityList="editEntityList"
        :originEntity="originEntity"
        @btnScript="runBtnScript"
        :processEntity="processEntity"

      <recycle-btn :entity="entity" :ids="ids" :btnsArr="buttons" v-show="isDelete == 1 && defaultBtn" @remove-refresh="removeRefresh" />
